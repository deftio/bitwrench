State & Interactivity

How bitwrench manages state, from simple closures to component factories and pub/sub.

Honest trade-off: Bitwrench does not have reactive data binding. When state changes, you call render() or bw.update() manually. This means you always know exactly when and why your UI updates — there is no hidden reactivity system. The trade-off is that you must remember to trigger updates yourself. React/Vue/Svelte handle this automatically; bitwrench trades convenience for simplicity.

The Simplest State Pattern

The most basic approach: a closure variable and a render function. When state changes, call render again. This works well for simple, standalone components.

How State Works in Bitwrench

In bitwrench, you build components by writing functions that return plain JavaScript objects describing what to render. State lives on the DOM element itself, not in separate variables or stores. When an element is removed from the page, its state goes with it. This avoids orphaned closures, stale references, and memory leaks.

1. Define o: { state: {...} }
→
2. Mount mounted(el)
→
3. Render bw.DOM(el, taco)

The pattern

A component factory returns a JavaScript object with o.state and o.mounted. When bitwrench creates the DOM element, it stores state as el._bw_state and calls mounted(el) once the element is in the DOM. The render function reads state from el._bw_state and calls bw.DOM(el, newContent) to re-render. State persists across re-renders because it's on the element, not in the content.

Why state lives on the element

If state were in a closure or global variable, removing the element from the DOM would leave the state floating in memory. By storing state on el._bw_state, the state is tied to the element's lifecycle. Delete the element → state is gone. Create three instances of a component → each has its own state on its own element. No registry to manage, no cleanup to remember.

The complete pattern
// 1. Factory function — returns a JS object, not a DOM element
function makeCounter(opts) {
  var id = bw.uuid('ctr');

  function render(el) {
    var s = el._bw_state;              // read state from the element
    bw.DOM(el, {                       // re-render into self
      t: 'div', a: { class: 'counter-card' },
      c: [
        { t: 'h4', c: opts.label },
        { t: 'div', c: String(s.count) },
        bw.makeButton({ text: '+',
          onclick: function() { s.count++; render(el); }
        })
      ]
    });
  }

  // 2. Return TACO with state and lifecycle hook
  return {
    t: 'div',
    a: { id: id },
    o: {
      state: { count: opts.start || 0 },  // initial state
      mounted: function(el) { render(el); }  // first render
    }
  };
}

// 3. Use it — each instance is independent
bw.DOM('#app', {
  t: 'div', c: [
    makeCounter({ label: 'A', start: 0 }),
    makeCounter({ label: 'B', start: 10 })
  ]
});
What happens at runtime
// bw.createDOM processes the TACO:
//   1. Creates <div id="ctr-abc123">
//   2. Stores state: el._bw_state = { count: 0 }
//   3. Queues mounted callback via requestAnimationFrame
//
// After the element is in the DOM:
//   4. mounted(el) fires → render(el) runs
//   5. render reads el._bw_state.count (= 0)
//   6. bw.DOM(el, newContent) replaces children
//      but preserves el._bw_state
//
// On button click:
//   7. s.count++ modifies el._bw_state.count
//   8. render(el) re-renders with new count
//
// If the element is removed:
//   9. el._bw_state goes away with the element
//   10. No cleanup needed — nothing external to clean up
//
// Key insight:
//   - opts (label, start) is configuration — fixed at creation
//   - state (count) is mutable — lives on the element
//   - render(el) is a pure function of el._bw_state

Encapsulated Components

Three counters from the same factory. Each has its own state on its own DOM element. Click any counter — the others don't change. This is the proof that state is truly encapsulated.

Source
function makeCounter(opts) {
  var id = bw.uuid('ctr');

  function render(el) {
    var s = el._bw_state;
    bw.DOM(el, {
      t: 'div', a: { class: 'counter-card' },
      c: [
        { t: 'h4', c: opts.label },
        { t: 'div', a: { class: 'counter-value' },
          c: String(s.count) },
        { t: 'div', a: { class: 'counter-btns' }, c: [
          bw.makeButton({ text: '+', variant: 'primary',
            onclick: function() { s.count++; render(el); }
          }),
          bw.makeButton({ text: '\u2013', variant: 'secondary',
            onclick: function() { s.count--; render(el); }
          }),
          bw.makeButton({ text: 'Reset', variant: 'secondary',
            onclick: function() {
              s.count = opts.start || 0; render(el);
            }
          })
        ]}
      ]
    });
  }

  return {
    t: 'div', a: { id: id },
    o: {
      state: { count: opts.start || 0 },
      mounted: function(el) { render(el); }
    }
  };
}

// 3 independent instances
bw.DOM('#counters', {
  t: 'div', a: { class: 'counters-row' },
  c: [
    makeCounter({ label: 'Visitors', start: 42 }),
    makeCounter({ label: 'Orders',   start: 7 }),
    makeCounter({ label: 'Errors',   start: 0 })
  ]
});

Stateful Component: Todo List

A more complex component with array state, conditional rendering, and multiple interactions. The same pattern scales: state on the element, render function reads it.

Source
function makeTodoList(opts) {
  var id = bw.uuid('todo');

  // Reusable TACO for a single todo item
  var todoItem = function(item, el) {
    var s = el._bw_state;
    return {
      t: 'div',
      a: { class: 'todo-item' + (item.done ? ' done' : '') },
      c: [
        { t: 'input', a: {
          type: 'checkbox',
          checked: item.done ? 'checked' : undefined,
          onchange: function() {
            item.done = !item.done; render(el);
          }
        }},
        { t: 'span', c: item.text },
        bw.makeButton({
          text: '\u00D7', variant: 'secondary', size: 'sm',
          onclick: function() {
            s.todos = s.todos.filter(function(t) {
              return t.id !== item.id;
            });
            render(el);
          }
        })
      ]
    };
  };

  function addTodo(el) {
    var s = el._bw_state;
    var input = bw.$('#' + id + ' input[type="text"]')[0];
    if (!input || !input.value.trim()) return;
    s.todos.push({
      id: s.nextId++,
      text: input.value.trim(),
      done: false
    });
    render(el);
  }

  function render(el) {
    var s = el._bw_state;
    var remaining = s.todos.filter(function(t) {
      return !t.done;
    }).length;

    bw.DOM(el, { t: 'div', c: [
      { t: 'div', a: { class: 'todo-input-row' }, c: [
        { t: 'input', a: {
          type: 'text',
          placeholder: opts.placeholder || 'Add item...',
          onkeydown: function(e) {
            if (e.key === 'Enter') addTodo(el);
          }
        }},
        bw.makeButton({
          text: 'Add', variant: 'primary',
          onclick: function() { addTodo(el); }
        })
      ]},
      { t: 'div', c: s.todos.map(function(item) {
        return todoItem(item, el);
      })},
      { t: 'div', a: {
        style: 'margin-top:0.5rem; font-size:0.8125rem; '
          + 'color:var(--bw-text-secondary);'
        },
        c: remaining + ' of ' + s.todos.length + ' remaining'
      }
    ]});
  }

  // Initial state: convert text strings to objects
  var initial = (opts.initial || []).map(function(text, i) {
    return { id: i + 1, text: text, done: false };
  });

  return {
    t: 'div', a: { id: id },
    o: {
      state: { todos: initial, nextId: initial.length + 1 },
      mounted: function(el) { render(el); }
    }
  };
}

bw.DOM('#todo', makeTodoList({
  initial: ['Read the docs', 'Build a prototype', 'Ship it'],
  placeholder: 'What needs doing?'
}));

Dynamic CSS Generation

bw.css() converts JavaScript objects to CSS strings. bw.injectCSS() inserts them into the document. Together they let you generate entire stylesheets at runtime from data.

Source
function makeThemeSwitcher() {
  var themes = {
    ocean:  { bg: '#e0f7fa', fg: '#004d40', accent: '#00796b' },
    sunset: { bg: '#fff3e0', fg: '#bf360c', accent: '#e65100' },
    forest: { bg: '#e8f5e9', fg: '#1b5e20', accent: '#2e7d32' }
  };
  var id = bw.uuid('theme');

  function render(el) {
    var s = el._bw_state;
    var t = themes[s.current];

    // Generate CSS from a JS object and inject it
    bw.injectCSS(bw.css({
      ['#' + id + ' .themed']: {
        background: t.bg, color: t.fg,
        'border-color': t.accent,
        padding: '1.25rem', 'border-radius': '8px',
        border: '2px solid', transition: 'all 0.3s'
      },
      ['#' + id + ' .themed h4']: {
        color: t.accent, margin: '0 0 0.35rem'
      }
    }), { id: id + '-css' });

    bw.DOM(el, { t: 'div', c: [
      { t: 'div',
        a: { style: 'display:flex; gap:0.5rem; margin-bottom:0.75rem;' },
        c: Object.keys(themes).map(function(key) {
          return bw.makeButton({
            text: key.charAt(0).toUpperCase() + key.slice(1),
            variant: key === s.current ? 'primary' : 'secondary',
            onclick: function() {
              s.current = key; render(el);
            }
          });
        })
      },
      { t: 'div', a: { class: 'themed' }, c: [
        { t: 'h4', c: s.current.charAt(0).toUpperCase()
          + s.current.slice(1) + ' Theme' },
        { t: 'p', a: { style: 'margin:0; font-size:0.875rem;' },
          c: 'Colors generated from a JS object via bw.css().' }
      ]}
    ]});
  }

  return {
    t: 'div', a: { id: id },
    o: {
      state: { current: 'ocean' },
      mounted: function(el) { render(el); }
    }
  };
}

bw.DOM('#themes', makeThemeSwitcher());

Performance

Bitwrench renders directly to the DOM with no virtual DOM diffing. This test creates and mounts 1,000 elements in a single bw.DOM() call.

Source
function makePerfTest() {
  var id = bw.uuid('perf');

  function run(el) {
    var start = performance.now();
    var items = [];
    for (var i = 0; i < 1000; i++) {
      items.push({
        t: 'span',
        a: { class: 'bw-badge bw-badge-primary',
             style: 'margin:2px;' },
        c: 'Item ' + (i + 1)
      });
    }
    bw.DOM('#' + id + '-items', { t: 'div', c: items });
    var ms = (performance.now() - start).toFixed(2);
    bw.DOM('#' + id + '-time', {
      t: 'div', a: { class: 'bw-alert bw-alert-success' },
      c: 'Rendered 1,000 items in ' + ms + ' ms'
    });
  }

  return {
    t: 'div', a: { id: id },
    o: {
      mounted: function(el) {
        bw.DOM(el, {
          t: 'div', c: [
            bw.makeButton({ text: 'Render 1,000 Items',
              variant: 'primary',
              onclick: function() { run(el); }
            }),
            { t: 'div', a: { id: id + '-time',
              style: 'margin-top:0.75rem;' }, c: '' },
            { t: 'div', a: { id: id + '-items',
              style: 'margin-top:0.75rem; max-height:200px; '
                + 'overflow-y:auto;' }, c: '' }
          ]
        });
      }
    }
  };
}

bw.DOM('#perf', makePerfTest());
